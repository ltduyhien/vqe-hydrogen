# Qiskit stack

# Core SDK: circuits, parameters, primitives (Estimator/Sampler), transpiler.

# Aer simulator. Needed for the noise sweep (density-matrix simulation).

# VQE algorithm + COBYLA/SPSA. Split out of qiskit after 0.x.

# Molecule -> qubit Hamiltonian. Used only by src/hamiltonian.py.
qiskit-nature>=0.7


# Classical chemistry driver

# Computes 1- and 2-electron integrals for qiskit-nature.
# Needs a working C/Fortran toolchain at install time.
pyscf>=2.3


# Scientific stack

numpy>=1.24

# Used internally by qiskit-algorithms optimizers.
scipy>=1.11

# Used only by src/plot.py.
# 3.8 is the first version whose axhline(label=...) works reliably on a
# log axis (needed in plot_noise_sweep).
matplotlib>=3.8
